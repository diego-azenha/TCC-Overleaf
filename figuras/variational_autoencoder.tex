%% ── COORDENADAS ──────────────────────────────────────────────────
%% Mesmos parâmetros do autoencoder: neurônio 0.52cm, espaçamento 0.85cm
%% Encoder: 5→4→3 | μ/σ (3+3) | Amostra z (3) | Decoder: 4→5

\foreach \i in {1,...,3}  \coordinate (SIG\i) at ( 5.8, {-0.30-(\i-1)*0.70});
\foreach \i in {1,...,3}  \coordinate (Z\i)   at ( 7.6, {(2.0-\i)*0.85});
\foreach \i in {1,...,4}  \coordinate (Da\i)  at ( 9.6, {(2.5-\i)*0.85});
\foreach \i in {1,...,5}  \coordinate (O\i)   at (11.6, {(3.0-\i)*0.85});

%% ── CONEXÕES (abaixo dos blocos e nós) ───────────────────────────
\foreach \i in {1,...,5} \foreach \j in {1,...,4} \draw[conn] (I\i)   -- (Ea\j);
\foreach \i in {1,...,4} \foreach \j in {1,...,3} \draw[conn] (Ea\i)  -- (Eb\j);
\foreach \i in {1,...,3} \foreach \j in {1,...,3} \draw[conn] (Eb\i)  -- (MU\j);
\foreach \i in {1,...,3} \foreach \j in {1,...,3} \draw[conn] (Eb\i)  -- (SIG\j);
\foreach \i in {1,...,3} \foreach \j in {1,...,3} \draw[conn] (MU\i)  -- (Z\j);
\foreach \i in {1,...,3} \foreach \j in {1,...,3} \draw[conn] (SIG\i) -- (Z\j);
\foreach \i in {1,...,3} \foreach \j in {1,...,4} \draw[conn] (Z\i)   -- (Da\j);
\foreach \i in {1,...,4} \foreach \j in {1,...,5} \draw[conn] (Da\i)  -- (O\j);

%% ── BLOCOS μ, σ, z (sobre conexões, abaixo dos nós) ─────────────
%% MU1=(5.8,1.70), MU3=(5.8,0.30) → caixa y=[0.05, 1.95]
\draw[VioletD, thick, rounded corners=4pt, fill=VioletL!55]

\draw[CoralD, thick, rounded corners=4pt, fill=CoralL!55]
    (5.50, -1.95) rectangle (6.10, -0.05);
\node[font=\small\bfseries, CoralD, anchor=north] at (5.80, -1.95) {Desvio-padrão $\sigma$};

%% Z1=(7.6,0.85), Z3=(7.6,−0.85) → caixa y=[−1.15, 1.15]
\draw[GreenD, thick, rounded corners=4pt, fill=GreenL!55]
    (7.28, -1.15) rectangle (7.92, 1.15);
\node[font=\small\bfseries, GreenD, anchor=south] at (7.60, 1.15) {Amostragem};

%% ── NÓS (sobre os blocos) ────────────────────────────────────────

    \node[neuron, fill=CoralL,  draw=CoralD]  at (SIG\i) {};
    \node[neuron, fill=GreenL,  draw=GreenD]  at (Z\i)   {};
