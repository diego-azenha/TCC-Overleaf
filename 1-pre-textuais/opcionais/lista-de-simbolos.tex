\begin{simbolos} \itemsep -1pt
  \item[$r_{i,t}$] Retorno do ativo $i$ no período $t$
  \item[$r_{t+1}$] Vetor de retornos no período $t+1$
  \item[$z_{t+1}$] Vetor de fatores latentes no período $t+1$
  \item[$\mathcal{F}_t$] Conjunto de informações disponíveis até o período $t$
  \item[$N_t$] Número de ativos disponíveis no período $t$
  \item[$F$] Número de fatores latentes
  \item[$\alpha_{i,t}$] Intercepto idiossincrático do ativo $i$ no período $t$
  \item[$\beta_{i,t}$] Vetor de exposições fatoriais do ativo $i$ no período $t$
  \item[$\sigma_{i,t}$] Parâmetro de escala do ativo $i$ no período $t$
  \item[$\nu_{i,t}$] Graus de liberdade da distribuição do ativo $i$ no período $t$
  \item[$\nu_z$] Graus de liberdade da distribuição prior dos fatores
  \item[$\mu_z$] Média prior dos fatores latentes
  \item[$\Sigma_z$] Matriz de covariância prior dos fatores latentes
  \item[$\Sigma_x$] Matriz diagonal de variâncias idiossincráticas
  \item[$B_t$] Matriz de exposições fatoriais no período $t$
  \item[$\mu_{z|B,r}$] Média posterior dos fatores latentes
  \item[$\Sigma_{z|B}$] Covariância posterior dos fatores latentes
  \item[$p(\cdot)$] Distribuição de probabilidade
  \item[$q(\cdot)$] Distribuição variacional aproximada
  \item[$K$] Número de amostras do IWAE
  \item[$w$] Vetor de pesos do portfólio
  \item[$\lambda$] Parâmetro de aversão ao risco
  \item[$S(\cdot)$] Função \textit{softplus}
  \item[$t_\nu(\mu, \sigma)$] Distribuição Student-$t$ com média $\mu$, escala $\sigma$ e $\nu$ graus de liberdade
  \item[$\mathcal{N}(\mu, \Sigma)$] Distribuição Normal com média $\mu$ e covariância $\Sigma$
\end{simbolos}
