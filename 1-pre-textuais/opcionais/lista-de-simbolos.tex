% lista-de-simbolos.tex
\begin{simbolos} \itemsep -1pt
  \item[$ \alpha $]           Componente idiossincrático esperado do retorno de um ativo
  \item[$ \beta $]            Matriz de exposições fatoriais
  \item[$ \varepsilon $]      Resíduo idiossincrático
  \item[$ \zeta $]            Coeficiente de inclinação negativa da LeakyReLU ($\zeta = 0{,}1$)
  \item[$ \mu $]              Média de uma distribuição gaussiana
  \item[$ \sigma $]           Desvio-padrão; também função sigmoide (equações da GRU)
  \item[$ \gamma $]           Peso do termo KL na função objetivo
  \item[$ \odot $]            Produto de Hadamard (multiplicação elemento a elemento)
  \item[$ \mathcal{L} $]      Função objetivo (negativo do ELBO)
  \item[$ \mathcal{N} $]      Distribuição normal (gaussiana)
  \item[$ \in $]              Pertence a (relação de pertencimento)
\end{simbolos}
